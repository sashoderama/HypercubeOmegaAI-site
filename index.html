      tok = chunk["choices"][0]["text"]; await on_token(tok)

class QuantumMemoryOptimizer:
  async def optimize(self, tensor):
    with autocast():
      f = torch.fft.fftn(tensor)              # hyper‑cube FFT
      return torch.real(f)/torch.norm(f)
